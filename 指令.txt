# 仅在“本地 STT 模式”下，首次准备 sherpa-sense-voice 本地模型（只需执行一次）
$modelUrl='https://github.com/k2-fsa/sherpa-onnx/releases/download/asr-models/sherpa-onnx-sense-voice-zh-en-ja-ko-yue-int8-2025-09-09.tar.bz2'
$tmp='D:\models\_tmp_sherpa_sense_voice'
$target='D:\models\sherpa-sense-voice'
New-Item -ItemType Directory -Force -Path $tmp | Out-Null
Invoke-WebRequest -Uri $modelUrl -OutFile "$tmp\model.tar.bz2"
tar -xjf "$tmp\model.tar.bz2" -C $tmp
$src = Get-ChildItem -Path $tmp -Recurse -Directory |
  Where-Object { (Test-Path "$($_.FullName)\model.int8.onnx") -and (Test-Path "$($_.FullName)\tokens.txt") } |
  Select-Object -First 1
if (-not $src) { throw '未找到 model.int8.onnx/tokens.txt，请检查下载是否完整' }
New-Item -ItemType Directory -Force -Path $target | Out-Null
Copy-Item "$($src.FullName)\model.int8.onnx" "$target\model.int8.onnx" -Force
Copy-Item "$($src.FullName)\tokens.txt" "$target\tokens.txt" -Force
